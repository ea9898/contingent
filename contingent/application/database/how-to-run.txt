#how to run

### Установка в схему
mvn liquibase:update -Poracle -Ddb.target.master=changelog/area/versions/master.xml -Ddb.target.url=jdbc:oracle:thin:@X.X.X.X:1521:emiasdb -Ddb.target.username=CONTINGENT -Ddb.target.password=CONTINGENT -Dliquibase.databaseChangeLogTableName=AREA_CHANGELOG -Dliquibase.databaseChangeLogLockTableName=AREA_CHANGELOGLOCK
mvn liquibase:update -Ppostgresql -Ddb.target.master=changelog/area/versions/master.xml -Ddb.target.url=jdbc:postgresql://X.X.X.X:5740/emiasdb -Ddb.target.username=CONTINGENT -Ddb.target.password=CONTINGENT -Dliquibase.databaseChangeLogTableName=AREA_CHANGELOG -Dliquibase.databaseChangeLogLockTableName=AREA_CHANGELOGLOCK

# В блоке rollback нужно указывать тег, к которому откатываемся. В нашем случае это будет версия, к которой нужно откатиться
java -jar liquibase.jar --driver=oracle.jdbc.OracleDriver
--classpath=ojdbc8.jar --changeLogFile=changelog/area/versions/master.xml
--url="jdbc:oracle:thin:@10.0.48.7:1521:emiasdb"
--username=CONTINGENT
--password=CONTINGENT
--logLevel=info
rollback 0.0.0

### Для разработки
mvn liquibase:update -Ppostgresql -Ddb.target.master=changelog/area/versions/master.xml -Ddb.target.url=jdbc:postgresql://dev.ptnl.moscow:5434/contingent2 -Ddb.target.username=contingent -Ddb.target.password=CONTINGENT -Dliquibase.databaseChangeLogTableName=AREA_CHANGELOG -Dliquibase.databaseChangeLogLockTableName=AREA_CHANGELOGLOCK
